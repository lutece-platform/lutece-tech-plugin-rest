package ${businessObject.packageName};

import fr.paris.lutece.portal.service.spring.SpringContextService;
import java.util.Collection;



/**
 * This class provides instances management methods (create, find, ...) for ${businessObject.className} objects
 */

public final class ${businessObject.className}Home
{

    // Static variable pointed at the DAO instance
    private static I${businessObject.className}DAO _dao = ( I${businessObject.className}DAO ) SpringContextService.getBean( "${businessObject.instanceName}DAO" );


    /**
     * Private constructor - this class need not be instantiated
     */

    private ${businessObject.className}Home(  )
    {
    }

    /**
     * Create an instance of the ${businessObject.instanceName} class
     * @param ${businessObject.instanceName} The instance of the ${businessObject.className} which contains the informations to store
     * @return The  instance of ${businessObject.instanceName} which has been created with its primary key.
     */

    public static ${businessObject.className} create( ${businessObject.className} ${businessObject.instanceName} )
    {
        _dao.insert( ${businessObject.instanceName} );

        return ${businessObject.instanceName};
    }


    /**
     * Update of the ${businessObject.instanceName} data specified in parameter
     * @param ${businessObject.instanceName} The instance of the ${businessObject.className} which contains the data to store
     * @return The instance of the  ${businessObject.instanceName} which has been updated
     */

    public static ${businessObject.className} update( ${businessObject.className} ${businessObject.instanceName} )
    {
        _dao.store( ${businessObject.instanceName} );

        return ${businessObject.instanceName};
    }


    /**
     * Remove the ${businessObject.instanceName} whose identifier is specified in parameter
     * @param n${businessObject.className}Id The ${businessObject.instanceName} Id
     */


    public static void remove( int n${businessObject.className}Id )
    {
        _dao.delete( n${businessObject.className}Id );
    }


    ///////////////////////////////////////////////////////////////////////////
    // Finders

    /**
     * Returns an instance of a ${businessObject.instanceName} whose identifier is specified in parameter   
     * @param nKey The ${businessObject.instanceName} primary key
     * @return an instance of ${businessObject.className}
     */

    public static ${businessObject.className} findByPrimaryKey( int nKey )
    {
        return _dao.load( nKey );
    }


    /**
     * Load the data of all the ${businessObject.instanceName} objects and returns them in form of a collection
     * @return the collection which contains the data of all the ${businessObject.instanceName} objects
     */

    public static Collection<${businessObject.className}> findAll()
    {
        return _dao.select${businessObject.className}sList();
    }

}
